11. Write JavaScript to calculate simple interest using user input values.



<!DOCTYPE html>
<html>
<head>
    <title>Simple Interest Calculator</title>
    <script>
        function calculateInterest() {
            // Get values from input fields
            let principal = Number(document.getElementById("principal").value);
            let rate = Number(document.getElementById("rate").value);
            let time = Number(document.getElementById("time").value);

            // Check for valid inputs
            if (isNaN(principal) || isNaN(rate) || isNaN(time) || principal < 0 || rate < 0 || time < 0) {
                document.getElementById("result").innerHTML = "Please enter valid positive numbers!";
                return;
            }

            // Simple Interest formula: SI = (P * R * T) / 100
            let simpleInterest = (principal * rate * time) / 100;

            // Display result
            document.getElementById("result").innerHTML = 
                `Simple Interest = ${simpleInterest.toFixed(2)}`;
        }
    </script>
</head>
<body>
    <h2>Simple Interest Calculator</h2>

    <input type="text" id="time" placeholder="Enter time"><br><br>

    <button onclick="calculateInterest()">Calculate</button>

            // Get numbers from input fields

            let num2 = Number(document.getElementById("num2").value);
            let num3 = Number(document.getElementById("num3").value);

                document.getElementById("result").innerHTML = "Please enter valid numbers!";
                return;
            }

            // Find the largest number
            let largest;
            if (num1 >= num2 && num1 >= num3) {
                largest = num1;
            } else if (num2 >= num1 && num2 >= num3) {
                largest = num2;
            } else {

        }

16. Validate a login form where:
 Username must be at least 5 characters
 Password must not be empty
 Error messages must show using JavaScript



<!DOCTYPE html>
<html>
<head>
    <title>Login Form Validation</title>
    <script>
        function validateForm() {
            // Get input values
            let username = document.getElementById("username").value;
            let password = document.getElementById("password").value;

            // Clear previous error messages
            document.getElementById("error").innerHTML = "";

            let messages = [];

            // Validate username (at least 5 characters)
            if (username.length < 5) {
                messages.push("Username must be at least 5 characters long.");
            }

            // Validate password (not empty)
            if (password === "") {
                messages.push("Password cannot be empty.");
            }

            // Display error messages or success
            if (messages.length > 0) {
                document.getElementById("error").innerHTML = messages.join("<br>");
                return false; // Prevent form submission
            } else {
                alert("Login successful!");
                return true; // Allow form submission
            }
        }
    </script>
</head>
<body>
    <h2>Login Form</h2>
    <form onsubmit="return validateForm()">
        <label>Username: </label>
        <input type="text" id="username" placeholder="Enter username"><br><br>

        <label>Password: </label>
        <input type="password" id="password" placeholder="Enter password"><br><br>

        <input type="submit" value="Login">
    </form>

    <p id="error" style="color: red;"></p>
</body>
</html>
